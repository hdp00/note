```
//npm升级
npm install npm -g
//安装模块
npm install <Module Name>
//-g 全局安装
//查看全局安装的模块
npm list -g
//卸载模块
npm uninstall <Module Name>
//更新模块
npm update <Module Name>

<//package.json>
//生成
npm init
//重新安装
npm install
//更新
npm update
//--save 写入到依赖
//--save-dev 写入到开发依赖
//--production,会下载依赖，不会下载开发依赖
```
